30   alignNeedlemanForTranscript_simd_avx2int32::alignNeedlemanForTranscript_simd_avx2int32(std::string& dna_d,
31                                                                                          std::string& dna_q, int & startCodonPosition, int & stopCodonPosition,
32                                                                                          std::vector<SpliceSitePosition>& splitSitePositions,
33                                                                                          std::map<std::string, std::string>& parameters, NucleotideCodeSubstitutionMatrix& nucleotideCodeSubstitutionMatrix){
34       //std::cout << "NeedlemanWunschForTranscript begin" << std::endl;
35       this->_dna_q = dna_q;
36       this->_dna_d = dna_d;
37       this->_exon_match_score = stoi(get_parameters("alignmentExonMatchP", parameters));
38       this->_exon_mismatch_score = stoi(get_parameters("alignmentExonMismatchP", parameters));
39   
40       this->_intron_match_score = stoi(get_parameters("alignmentIntronMatchP", parameters));
41       this->_intron_mismatch_score  = stoi(get_parameters("alignmentIntronMismatchP", parameters));
42   
43       this->_splice_sites_match_score = stoi(get_parameters("alignmentSpliceSitesMatchP", parameters));
44       this->_splice_sites_mismatch_score = stoi(get_parameters("alignmentSpliceSitesMismatchP", parameters));
45   
46       this->_start_stop_codon_match_score = stoi(get_parameters("alignmentStartStopCodonMatchP", parameters));
47       this->_start_stop_codon_mismatch_score = stoi(get_parameters("alignmentStartStopCodonMismatchP", parameters));
48   
49       _open_gap_penalty = new int32_t[4];
50       _extend_gap_penalty = new int32_t[4];
51       _open_gap_penalty[0]=stoi(get_parameters("alignmentExonOpenGapP", parameters));
52       _extend_gap_penalty[0]=stoi(get_parameters("alignmentExonExtendGapP", parameters));
53   
54       _open_gap_penalty[1]=stoi(get_parameters("alignmentIntronOpenGapP", parameters));
55       _extend_gap_penalty[1]=stoi(get_parameters("alignmentIntronExtendGapP", parameters));
56   
57       _open_gap_penalty[2]=stoi(get_parameters("alignmentSpliceSitesOpenGapP", parameters));
58       _extend_gap_penalty[2]=stoi(get_parameters("alignmentSpliceSitesExtendGapP", parameters));
59   
60       _open_gap_penalty[3]=stoi(get_parameters("alignmentStartStopCodonOpenGapP", parameters));
61       _extend_gap_penalty[3]=stoi(get_parameters("alignmentStartStopCodonExtendGapP", parameters));
62   
63       this->_startCodonPosition = startCodonPosition;
64       this->_stopCodonPosition = stopCodonPosition;
65       this->_spliceSitePositions = splitSitePositions;
66   
67       this->_length_of_q = this->_dna_q.length();
68       this->_length_of_d = this->_dna_d.length();
69   
70       this->bias = 0;
71       this->_n = 5;
72       this->_segLen = ( _length_of_q+ 7)/8;
73   
74       //check potential overflow problem begin
75       int32_t maxLength = _length_of_d;
76       if( maxLength <  _length_of_q){
77           maxLength = _length_of_q;
78       }
79       int32_t maxScore = _exon_match_score;
80       if( maxScore < _intron_match_score ){
81           maxScore = _intron_match_score;
82       }
83       if( maxScore < _splice_sites_match_score ){
84           maxScore = _splice_sites_match_score;
85       }
86       if( maxScore < _start_stop_codon_match_score ){
87           maxScore = _start_stop_codon_match_score;
88       }
89       if( (maxLength * maxScore) > pow(2,31)){
90           std::cerr << "the sequence is too long, could not be optimized with this pipeline" << std::endl;
91           return;
92       }
93       int32_t minimumPenalty = _exon_mismatch_score;
94   
95       if( minimumPenalty > _intron_mismatch_score ){
96           minimumPenalty = _intron_mismatch_score;
97       }
98       if( minimumPenalty > _splice_sites_mismatch_score ){
99           minimumPenalty = _splice_sites_mismatch_score;
100      }
101      if( minimumPenalty > _start_stop_codon_mismatch_score ){
102          minimumPenalty = _start_stop_codon_mismatch_score;
103      }
104      int32_t i, j;
105      for( i=0; i<4; ++i ){
106          if( minimumPenalty > _open_gap_penalty[i] ){
107              minimumPenalty =  _open_gap_penalty[i];
108          }
109      }
110      if( (maxLength * minimumPenalty) < -pow(2,31)){
111          std::cerr << "the sequence is too long, could not be optimized with this pipeline" << std::endl;
112          return;
113      } //check potential overflow problem end
114  
115      for( i=0; i<4; ++i ){
116          if( _open_gap_penalty[i] > _extend_gap_penalty[i] ){
117              std::cerr << "it is not reasonable to have open gap penalty larger than extend gap penalty" << std::endl;
118          }
119      }//warning message about score strategy end
120  
121      this->_similarity_matrix = new int32_t[this->_length_of_q + 1];
122      for ( i = 0; i<= _length_of_q; ++i) {
123          _similarity_matrix[i] = 0;
124      }
125      this->_trace_match = new std::vector<bool> [this->_length_of_q + 1];
126      this->_trace_del = new std::vector<bool> [this->_length_of_q + 1];
127      this->_trace_ins = new std::vector<bool> [this->_length_of_q + 1];
128      for ( i = 0; i < (this->_length_of_q + 1); ++i) {
129          _trace_match[i].resize(_length_of_d+1);
130          _trace_del[i].resize(_length_of_d+1);
131          _trace_ins[i].resize(_length_of_d+1);
132      }
133  
134      for ( i = 0; i <= 1; i++) {
135          for ( j = 0; j <= _length_of_d; j++) {
136              _trace_match[i][j] = true;
137              _trace_del[i][j] = true;
138              _trace_ins[i][j] = true;
139          }
140      }
141      for ( j = 0; j <= 1; j++){
142          for ( i = 0; i <= _length_of_q; i++) {
143              _trace_match[i][j] = true;
144              _trace_del[i][j] = true;
145              _trace_ins[i][j] = true;
146          }
147      }
148      _ref_elements = new int8_t[this->_length_of_d];
149      for (i = 0; i<_length_of_d; ++i) {//important
150          _ref_elements[i]=this->checkElements(i + 1); // because the gff records is start from 1
151      }
152      //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 151" << std::endl;
153      __m256i** vProfile = query_profile_avx2_byte(nucleotideCodeSubstitutionMatrix);
154      //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 153" << std::endl;
155      this->calculate_similarity(vProfile, nucleotideCodeSubstitutionMatrix);
156      //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 155" << std::endl;
157      this->get_optimal_alignment(nucleotideCodeSubstitutionMatrix);
158      //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 157" << std::endl;
159     // this->print_results();
160  }

172  int alignNeedlemanForTranscript_simd_avx2int32::checkElements( int position ){
173      if( position < this->_startCodonPosition || position > this->_stopCodonPosition+2  ){
174          return 1;//INTRON;
175      }
176      for(std::vector<SpliceSitePosition>::size_type i = 0; i != this->_spliceSitePositions.size(); i++ ){
177          SpliceSitePosition spliceSite = this->_spliceSitePositions[i];
178          if(position > (spliceSite.getDonorSpliceSitePosition()+1) && position < (spliceSite.getAcceptorSpliceSitePosition()-1) ){
179              return 1;//INTRON;
180          }else if( position== spliceSite.getAcceptorSpliceSitePosition() || position == spliceSite.getAcceptorSpliceSitePosition()-1){
181              return 2;//SPLICEACCEPTOR;
182          }else if( position== spliceSite.getDonorSpliceSitePosition() || position == spliceSite.getDonorSpliceSitePosition()+1  ){
183              return 2;//SPLICEDONOR;
184          }
185      }
186      if (position == this->_startCodonPosition || position == this->_startCodonPosition+1 || position == this->_startCodonPosition+2 ){
187          return 3;//START;
188      }
189      if (position == this->_stopCodonPosition || position == this->_stopCodonPosition+1 || position == this->_stopCodonPosition+2 ){
190          return 3;//STOP;
191      }
192      return 0;//EXON;
193  }

195  __m256i** alignNeedlemanForTranscript_simd_avx2int32::query_profile_avx2_byte(
196          NucleotideCodeSubstitutionMatrix& nucleotideCodeSubstitutionMatrix){ // dna_q is the query, (the non-reference sequence, with gene structure unknown)
197      __m256i* exon_profile = (__m256i*)malloc(this->_n * this->_segLen * sizeof(__m256i));
198      __m256i* intron_profile = (__m256i*)malloc(this->_n * this->_segLen * sizeof(__m256i));
199      __m256i* splice_sites_profile = (__m256i*)malloc(this->_n * this->_segLen * sizeof(__m256i));
200      __m256i* start_stop_profile = (__m256i*)malloc(this->_n * this->_segLen * sizeof(__m256i));
201  
202      int32_t a, i, j, k;
203      for( a =0; a<_n; ++a ){
204          //vProfile[a] = new __m256i[segLen];
205          int32_t* exon_s = (int32_t*)(exon_profile+a*this->_segLen);
206          int32_t* intron_s = (int32_t*)(intron_profile+a*this->_segLen);
207          int32_t* splice_sites_s = (int32_t*)(splice_sites_profile+a*this->_segLen);
208          int32_t* start_stop_s = (int32_t*)(start_stop_profile+a*this->_segLen);
209          for( i=0; i<this->_segLen; ++i ){
210              j = i;
211              for( k=0; k<8; ++k ){
212                  *exon_s++ = j>= _length_of_q ? bias : nucleotideCodeSubstitutionMatrix.get_exon_subsitition_matrix()[a][nucleotideCodeSubstitutionMatrix.get_dna_acid_map(_dna_q[j])];
213                  *intron_s++ = j>= _length_of_q ? bias : nucleotideCodeSubstitutionMatrix.get_intron_subsitition_matrix()[a][nucleotideCodeSubstitutionMatrix.get_dna_acid_map(_dna_q[j])];
214                  *splice_sites_s++ = j>= _length_of_q ? bias : nucleotideCodeSubstitutionMatrix.get_splice_sites_subsitition_matrix()[a][nucleotideCodeSubstitutionMatrix.get_dna_acid_map(_dna_q[j])];
215                  *start_stop_s++ = j>= _length_of_q ? bias : nucleotideCodeSubstitutionMatrix.get_start_stop_codon_subsitition_matrix()[a][nucleotideCodeSubstitutionMatrix.get_dna_acid_map(_dna_q[j])];
216                  j += this->_segLen;
217              }
218          }
219      }
220      __m256i** vProfile =  new __m256i*[4];
221      vProfile[0] = exon_profile;
222      vProfile[1] = intron_profile;
223      vProfile[2] = splice_sites_profile;
224      vProfile[3] = start_stop_profile;
225      return vProfile;
226  }

229  void alignNeedlemanForTranscript_simd_avx2int32::calculate_similarity( __m256i** vProfile,
230                                                                         NucleotideCodeSubstitutionMatrix& nucleotideCodeSubstitutionMatrix){
231      //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 229" << std::endl;
232      const __m256i vZero = _mm256_set1_epi32(0);
233      const __m256i minimum = _mm256_set1_epi32(-1147483648);
234      //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 231" << std::endl;
235      __m256i* pvHStore = (__m256i*) calloc(this->_segLen, sizeof(__m256i));
236      __m256i* pvHLoad = (__m256i*) calloc(this->_segLen, sizeof(__m256i));
237      __m256i* pvE = (__m256i*) calloc(this->_segLen, sizeof(__m256i));
238      __m256i* pvEC = (__m256i*) calloc(this->_segLen, sizeof(__m256i));
239      __m256i* pvFC = (__m256i*) calloc(this->_segLen, sizeof(__m256i));
240      __m256i* pvMC = (__m256i*) calloc(this->_segLen, sizeof(__m256i));
241      __m256i* pv;
242      __m256i* vP;
243      __m256i vF;
244      __m256i e;
245      __m256i vGapO, vGapE, next_vGapO, next_vGapE, vTemp;
246      //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 242" << std::endl;
247      int z;
248      int32_t i, j, k, cmp;
249      int32_t * a;
250      int32_t * b;
251      int32_t * c;
252      int32_t * d;
253      //for outer loop
254      __m256i vH;
255      for (i = 0; i<_length_of_d; ++i) {//important
256          //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 251" << std::endl;
257          vF = minimum; /* Initialize F value to 0. 
258                                 Any errors to vH values will be corrected in the Lazy_F loop. 
259                               */
260          a = ( int32_t*)(&vF);
261          a[0] = _extend_gap_penalty[_ref_elements[i]];
262          //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 255" << std::endl;
263          vP = vProfile[_ref_elements[i]] + ((nucleotideCodeSubstitutionMatrix.get_dna_acid_map(_dna_d[i]))*(this->_segLen)); // Correct part of the vProfile
264          //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 257" << std::endl;
265          // INDEL begin vector
266          vGapO = _mm256_set1_epi32(_open_gap_penalty[_ref_elements[i]]); // here I use epi32, because for the score matrix I used int32_t
267          // INDEL extension vector
268          vGapE = _mm256_set1_epi32(_extend_gap_penalty[_ref_elements[i]]);
269          if( i == _length_of_d-1){
270              next_vGapO=vGapO;
271              next_vGapE=vGapE;
272          }else{
273              next_vGapO=_mm256_set1_epi32(_open_gap_penalty[_ref_elements[i+1]]);
274              next_vGapE= _mm256_set1_epi32(_extend_gap_penalty[_ref_elements[i+1]]);
275          }
276          //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 269" << std::endl;
277          vH = _mm256_loadu_si256(pvHStore+(this->_segLen-1));// the last one
278          //vH = pvHStore[_segLen - 1];
279          //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 271" << std::endl;
280          vH = _mm256_alignr_epi8(vH, _mm256_permute2x128_si256(vH, vH, _MM_SHUFFLE(0, 0, 2, 0)), 16 - 4);
281  
282          //vMaxColumn = vZero; /* vMaxColumn is used to record the max values of column i. */
283          pv = pvHLoad;
284          pvHLoad = pvHStore;
285          pvHStore = pv;
286          //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 280" << std::endl;
287          // inner loop to process the query sequence
288          for (j = 0; j < this->_segLen; ++j) {
289              if( 0 == i ){
290                  e = vGapE;
291              }else{
292                  e = _mm256_loadu_si256(pvE + j);
293              }
294              vH = _mm256_add_epi32(vH, _mm256_loadu_si256(vP+j)); // add the scoring profile to vH
295                  _mm256_storeu_si256(pvEC+j, e);
296                  _mm256_storeu_si256(pvMC+j, vH);
297              vH = _mm256_max_epi32(vH, e);
298              vH = _mm256_max_epi32(vH, vF);
299              //vMaxColumn = _mm256_max_epi32(vMaxColumn, vH);
300  
301              /* Save vH values. */
302              _mm256_storeu_si256(pvHStore + j, vH);
303              /* Update vE value. */
304              vH = _mm256_add_epi32(vH, next_vGapO); // this is for next column
305              e = _mm256_add_epi32(e, next_vGapE);
306              e = _mm256_max_epi32(e, vH);
307              _mm256_storeu_si256(pvE + j, e);
308  
309              _mm256_storeu_si256(pvFC+j, vF);
310  
311              /* Update vF value. */
312              vH = _mm256_loadu_si256(pvHStore + j);
313              vH = _mm256_add_epi32(vH, vGapO);
314              vF = _mm256_add_epi32(vF, vGapE);
315              vF = _mm256_max_epi32(vF, vH);
316  
317              /* Load the next vH. */
318              vH = _mm256_loadu_si256(pvHLoad + j);
319          }
320          /* Lazy_F loop: has been revised to disallow adjecent insertion and then deletion, so don't update E(i, j), learn from SWPS3 */
321          /* reset pointers to the start of the saved data */
322          j=0;
323          vH = _mm256_loadu_si256 (pvHStore + j);
324          vF = _mm256_alignr_epi8(vF, _mm256_permute2x128_si256(vF, vF, _MM_SHUFFLE(0, 0, 2, 0)), 16 - 4); //shift 4 byte (int32)
325          a = ( int32_t*)(&vF);
326          a[0] = _extend_gap_penalty[_ref_elements[i]];
327          _mm256_storeu_si256(pvFC+j, vF);
328          vTemp = _mm256_add_epi32 (vH, vGapO);
329          vTemp = _mm256_sub_epi32 (vF, vTemp);
330  //        vTemp = _mm256_cmpeq_epi32 (vTemp, vZero);
331  //        cmp  = _mm256_movemask_epi8 (vTemp);
332          a = (int32_t *) (&vTemp);
333          //std::cout << "cmp " << cmp << std::endl;
334          while ( a[0]>0 || a[1]>0 || a[2]>0 || a[3]>0 || a[4]>0 || a[5]>0 || a[6]>0 || a[7]>0){
335                  //std::cout << "line 321 cmp " << cmp << std::endl;
336              vH = _mm256_max_epi32 (vH, vF);
337              _mm256_storeu_si256 (pvHStore + j, vH);
338              vF = _mm256_add_epi32 (vF, vGapE);
339              ++j;
340              if (j >= this->_segLen){
341                  j = 0;
342                  vF = _mm256_alignr_epi8(vF, _mm256_permute2x128_si256(vF, vF, _MM_SHUFFLE(0, 0, 2, 0)), 16 - 4); //shift 4 byte (int32)
343                  a = ( int32_t*)(&vF);
344                  a[0] = _extend_gap_penalty[_ref_elements[i]];
345              }
346                  _mm256_storeu_si256(pvFC+j, vF);
347              vH = _mm256_loadu_si256 (pvHStore + j);
348              vTemp = _mm256_add_epi32 (vH, vGapO);
349              vTemp = _mm256_sub_epi32 (vF, vTemp);
350              a = (int32_t *) (&vTemp);
351  //            vTemp = _mm256_cmpeq_epi32 (vTemp, vZero);
352   //           cmp  = _mm256_movemask_epi8 (vTemp);
353          }
354  
355          //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 320" << std::endl;
356          for( j=0; j<this->_segLen; ++j ){
357              a = ( int32_t*)(pvHStore+j);
358              b = ( int32_t*)(pvMC+j);
359              c = ( int32_t*)(pvEC+j);
360              //d = ( int32_t*)(pvFC+j);
361              for( z=0; z<8; ++z ){
362                  if( (j + z*this->_segLen) < _length_of_q ){
363                      //std::cout << j+z*this->_segLen+1 << " " << i+1 << " " << a[z] << " " << b[z] << " " << c[z] << std::endl;
364                      _trace_match[j+z*this->_segLen+1][i+1] = ( a[z] == b[z]);
365                      _trace_del[j+z*this->_segLen+1][i+1] = ( a[z] == c[z] );
366                //      _trace_ins[j+z*this->_segLen+1][i+1] = ( a[z] == d[z] );
367  
368                      if( _trace_del[j+z*this->_segLen+1][i+1] && !_trace_match[j+z*this->_segLen+1][i+1] /*&& !_trace_ins[j+z*this->_segLen+1][i+1] */ ){
369                          for ( int hi=i; hi>=0; --hi ){
370                              if( _trace_del[j+z*this->_segLen+1][hi] ){
371                                  if(_trace_match[j+z*this->_segLen+1][hi]){
372                                      _trace_match[j+z*this->_segLen+1][hi] = false;
373                                  }
374                              }else{
375                                  break;
376                              }
377                          }
378                          /* 
379                          for ( int hi=i; hi>=0; --hi ){ 
380                              if( _trace_del[j+z*this->_segLen+1][hi] ){ 
381                                  if( _trace_ins[j+z*this->_segLen+1][hi] ) { 
382                                      _trace_ins[j + z * this->_segLen + 1][hi] = false; 
383                                  } 
384                              }else{ 
385                                  break; 
386                              } 
387                          }*/
388                      }
389                      /*if( _trace_ins[j+z*this->_segLen+1][i+1]  && !_trace_match[j+z*this->_segLen+1][i+1] && !_trace_del[j+z*this->_segLen+1][i+1] ){ 
390                          for ( int hi=j+z*this->_segLen; hi>=0; --hi ){ 
391                              if( _trace_ins[hi][i+1]){ 
392                                  if(_trace_match[hi][i+1] ) { 
393                                      _trace_match[hi][i + 1] = false; 
394                                  } 
395                              }else{ 
396                                  break;

403                                  }

407                          } 
408                      }*/
409                  }
410              }
411          }
412          //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 339" << std::endl;
413      }
414      //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 341" << std::endl;
415      for( j=0; j<this->_segLen; ++j ){
416          a = ( int32_t*)(pvHStore+j);
417          for( z=0; z<8; ++z ){
418              if( (j + z*this->_segLen) < _length_of_q ){
419                  _similarity_matrix[j+z*this->_segLen+1] = a[z];
420              }
421          }
422      }
423      //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 335" << std::endl;
424      free(pvHStore);
425      //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 421" << std::endl;
426      free(pvHLoad);
427      //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 423" << std::endl;
428      free(pvE);
429      //std::cout << "alignNeedlemanForTranscript_simd_avx2int32 line 425" << std::endl;
430      free(pvEC);

451  void alignNeedlemanForTranscript_simd_avx2int32::get_optimal_alignment( NucleotideCodeSubstitutionMatrix& _nucleotideCodeSubstitutionMatrix ) {
452      _alignment_q = "";
453      _alignment_d = "";
454      _infor = "";
455      std::stack<char> SQ, SD;
456      std::stack<char> SI;
457  
458      int32_t k;
459      size_t i = this->_length_of_q;
460      size_t j = this->_length_of_d;
461      int32_t highestScore = this->_similarity_matrix[i];
462      for( k=_length_of_q; k >0; --k ){
463          if( _similarity_matrix[k] > highestScore ){
464              i = k;
465              highestScore= _similarity_matrix[k];
466          }
467      }
468      for( k=_length_of_q; k>i; --k ){
469          SQ.push(_dna_q[k-1]);
470          SD.push('-');
471          SI.push('-');
472      }
473      while (i > 0 || j > 0) {
474          char element = element_int_to_char(this->_ref_elements[j - 1]);
475          if (i == 0) {
476              SQ.push('-');
477              SD.push(_dna_d[j - 1]);
478              SI.push(element);
479              --j;
480          } else if (j == 0) {
481              SQ.push(_dna_q[i - 1]);
482              SD.push('-');
483              SI.push('-');
484              --i;
485          } else {
486              if (_trace_match[i][j]) {
487                  SQ.push(_dna_q[i - 1]);
488                  SD.push(_dna_d[j - 1]);
489                  SI.push(element);
490                  --i;
491                  --j;
492              } else if (_trace_del[i][j]) {//insertion
493                  SQ.push('-');
494                  SD.push(_dna_d[j - 1]);
495                  SI.push(element);
496                  --j;
497              } else {        // Going to S(i, j-1) //deletion
498                  SQ.push(_dna_q[i - 1]);
499                  SD.push('-');
500                  SI.push('-');
501                  --i;
502              }
503          }
504      }
505      while (!SQ.empty()) {
506          _alignment_q += SQ.top();
507          _alignment_d += SD.top();
508          _infor += SI.top();
509          SQ.pop();
510          SD.pop();
511          SI.pop();
512      }
513  }
